    def load_json(self, path, default):
        if os.path.exists(path):
            try:
                with open(path, 'r', encoding='utf-8') as f:
                    return json.load(f)
            except:
                return default
        return default

    def save_all(self):
        """Salvează totul automat."""
        with open(self.data_file, 'w', encoding='utf-8') as f:
            json.dump(self.data, f, indent=2, ensure_ascii=False)
        with open(self.catalog_file, 'w', encoding='utf-8') as f:
            json.dump(self.catalog, f, indent=2, ensure_ascii=False)
        # Actualizează summary când salvăm
        self.update_summary()

    def update_summary(self):
        """Actualizează panoul de sumar cu datele din luna curentă."""
        m_idx = list(calendar.month_name).index(self.month_var.get())
        month_key = f"{self.year_var.get()}-{m_idx:02d}"
        
        # Titlu
        self.lbl_summary_title.config(text=f"{self.month_var.get()} {self.year_var.get()}")
        
        total_income = 0
        total_expense = 0
        work_days = 0
        total_entries = 0
        
        if month_key in self.data:
            month_data = self.data[month_key]
            for day_key, day_data in month_data.items():
                if isinstance(day_data, dict):
                    # Date financiare
                    income = float(day_data.get('income', 0) or 0)
                    expense = float(day_data.get('expense', 0) or 0)
                    total_income += income
                    total_expense += expense
                    
                    # Intrări
                    entries = day_data.get('entries', [])
                    if entries or income > 0 or expense > 0:
                        work_days += 1
                    total_entries += len(entries)
                elif isinstance(day_data, list):
                    # Format vechi - doar intrări
                    if day_data:
                        work_days += 1
                    total_entries += len(day_data)
        
        profit = total_income - total_expense
        avg_profit = profit / work_days if work_days > 0 else 0
        
        # Actualizare labels
        self.lbl_total_income.config(text=f"{total_income:,.2f} Lei")
        self.lbl_total_expense.config(text=f"{total_expense:,.2f} Lei")
        self.lbl_total_profit.config(text=f"{profit:,.2f} Lei")
        self.lbl_work_days.config(text=str(work_days))
        self.lbl_total_entries.config(text=str(total_entries))
        self.lbl_avg_profit.config(text=f"{avg_profit:,.2f} Lei")
        
        # Culoare profit
        if profit >= 0:
            self.lbl_total_profit.config(fg='#2E7D32')
        else:
            self.lbl_total_profit.config(fg='#C62828')

    def save_finance(self, event=None):
        """Salvează datele financiare pentru ziua curentă."""
        if not hasattr(self, 'current_day'):
            return
            
        mk, dk = self.get_key()
        if mk not in self.data:
            self.data[mk] = {}
        
        # Migrează la noul format dacă e necesar
        if dk not in self.data[mk]:
            self.data[mk][dk] = {'income': 0, 'expense': 0, 'entries': []}
        elif isinstance(self.data[mk][dk], list):
            # Migrează din formatul vechi
            old_entries = self.data[mk][dk]
            self.data[mk][dk] = {'income': 0, 'expense': 0, 'entries': old_entries}
        
        # Salvează valorile
        try:
            self.data[mk][dk]['income'] = float(self.entry_income.get() or 0)
        except ValueError:
            self.data[mk][dk]['income'] = 0
            
        try:
            self.data[mk][dk]['expense'] = float(self.entry_expense.get() or 0)
        except ValueError:
            self.data[mk][dk]['expense'] = 0
        
        # Calculează și afișează profitul zilnic
        income = self.data[mk][dk]['income']
        expense = self.data[mk][dk]['expense']
        daily_profit = income - expense
        
        self.lbl_daily_profit.config(text=f"{daily_profit:,.2f} Lei")
        if daily_profit >= 0:
            self.lbl_daily_profit.config(fg='#2E7D32')
        else:
            self.lbl_daily_profit.config(fg='#C62828')
        
        self.save_all()

    def load_finance_for_day(self):
        """Încarcă datele financiare pentru ziua selectată."""
        if not hasattr(self, 'current_day'):
            return
            
        mk, dk = self.get_key()
        
        # Resetează câmpurile
        self.entry_income.delete(0, tk.END)
        self.entry_expense.delete(0, tk.END)
        
        if mk in self.data and dk in self.data[mk]:
            day_data = self.data[mk][dk]
            if isinstance(day_data, dict):
                income = day_data.get('income', 0)
                expense = day_data.get('expense', 0)
                
                if income:
                    self.entry_income.insert(0, str(income))
                if expense:
                    self.entry_expense.insert(0, str(expense))
                
                daily_profit = income - expense
                self.lbl_daily_profit.config(text=f"{daily_profit:,.2f} Lei")
                if daily_profit >= 0:
                    self.lbl_daily_profit.config(fg='#2E7D32')
                else:
                    self.lbl_daily_profit.config(fg='#C62828')
            else:
                self.lbl_daily_profit.config(text="0.00 Lei", fg='#1565C0')
        else:
            self.lbl_daily_profit.config(text="0.00 Lei", fg='#1565C0')

            def filter_mats(event):

                name = combo.get().strip()
                qty = ent_qty.get().strip()
                
                if name and qty:
                    # Verificăm dacă există deja în catalog
                    found_unit = None
                    for m in self.catalog:
                        if m['nume'].lower() == name.lower():
                            found_unit = m['unit']
                            name = m['nume']  # Corectăm capitalizarea
                            break

                                                        f"Materialul '{name}' este nou!\nCe unitate de măsură are? (m, buc, kg etc.)",

                            return

                    messagebox.showwarning("Lipsă date", "Scrieți numele materialului și cantitatea.")

    # --- CALCULATOR ---
    def build_calculator(self, parent):
        f = ttk.LabelFrame(parent, text="🔢 Calculator", padding=10)
        f.pack(fill=tk.X, padx=5, pady=10)
        
        self.calc_display = ttk.Entry(f, font=('Consolas', 14), justify='right')
        self.calc_display.pack(fill=tk.X, pady=8)
        
        btns_frame = ttk.Frame(f)
        btns_frame.pack()
        
        buttons = ['7', '8', '9', '/', '4', '5', '6', '*', '1', '2', '3', '-', 'C', '0', '=', '+']
        r, c = 0, 0
        for b in buttons:
            cmd = lambda x=b: self.calc_click(x)
            bg_col = '#E3F2FD' if b.isdigit() or b == '.' else '#BBDEFB'
            if b == 'C':
                bg_col = '#ffcdd2'
            elif b == '=':
                bg_col = '#C8E6C9'
            tk.Button(btns_frame, text=b, width=5, height=2, command=cmd, 
                      bg=bg_col, font=('Segoe UI', 11, 'bold'), cursor='hand2').grid(row=r, column=c, padx=2, pady=2)
            c += 1
            if c > 3:
                c = 0
                r += 1

    def calc_click(self, key):
        if key == 'C':
            self.calc_display.delete(0, tk.END)
        elif key == '=':
            try:
                res = eval(self.calc_display.get())
                self.calc_display.delete(0, tk.END)
                self.calc_display.insert(tk.END, str(res))
            except:
                messagebox.showerror("Eroare", "Calcul greșit")
        else:
            self.calc_display.insert(tk.END, key)

    # --- HELPERS ---
    def format_materials(self, entry):
        lst = entry.get('lista_mat', [])
        if lst:
            return "\n".join([f"• {x['n']} {x['q']}{x['u']}" for x in lst])
        return ""

    def load_month(self):
        self.listbox_days.delete(0, tk.END)
        y, m = int(self.year_var.get()), list(calendar.month_name).index(self.month_var.get())
        _, days = calendar.monthrange(y, m)
        for d in range(1, days + 1):
            self.listbox_days.insert(tk.END, f"  📅 Ziua {d}")
        self.update_summary()

    def select_day(self, event):
        sel = self.listbox_days.curselection()
        if sel:
            self.current_day = sel[0] + 1
            self.lbl_day_info.config(text=f"📅 Activitate: {self.current_day} {self.month_var.get()} {self.year_var.get()}")
            self.load_finance_for_day()
            self.refresh_entries_ui()

    def get_key(self):
        m_idx = list(calendar.month_name).index(self.month_var.get())
        return f"{self.year_var.get()}-{m_idx:02d}", str(self.current_day)

    def add_entry(self, tip):
        if not hasattr(self, 'current_day'):
            messagebox.showwarning("!", "Selectează o zi din stânga!")
            return
        mk, dk = self.get_key()
        if mk not in self.data:
            self.data[mk] = {}
        
        # Migrează la noul format dacă e necesar
        if dk not in self.data[mk]:
            self.data[mk][dk] = {'income': 0, 'expense': 0, 'entries': []}
        elif isinstance(self.data[mk][dk], list):
            old_entries = self.data[mk][dk]
            self.data[mk][dk] = {'income': 0, 'expense': 0, 'entries': old_entries}
        
        self.data[mk][dk]['entries'].append({"Tip": tip, "Locatie": "", "Comentariu": "", "Persoane": ""})
        self.save_all()
        self.refresh_entries_ui()

    def delete_entry(self, idx):
        if messagebox.askyesno("Șterge", "Sigur ștergi această intrare?"):
            mk, dk = self.get_key()
            day_data = self.data[mk][dk]
            if isinstance(day_data, dict):
                del day_data['entries'][idx]
            else:
                del day_data[idx]
            self.save_all()
            self.refresh_entries_ui()

    def refresh_entries_ui(self):
        for w in self.scrollable_frame.winfo_children():
            w.destroy()
        mk, dk = self.get_key()
        day_data = self.data.get(mk, {}).get(dk, {})
        
        if isinstance(day_data, dict):
            entries = day_data.get('entries', [])
        else:
            entries = day_data  # Format vechi
            
        for i, ent in enumerate(entries):
            self.draw_entry_card(i, ent)

    def manual_save(self):
        self.save_all()
        messagebox.showinfo("💾 Salvat", "Datele au fost salvate cu succes!")

    def export_excel(self):
        rows = []
        for mk, days in self.data.items():
            for dk, day_data in days.items():
                if isinstance(day_data, dict):
                    income = day_data.get('income', 0)
                    expense = day_data.get('expense', 0)
                    entries = day_data.get('entries', [])
                else:
                    income = 0
                    expense = 0
                    entries = day_data
                
                # Adaugă rând pentru finanțe
                if income > 0 or expense > 0:
                    rows.append({
                        "Data": f"{dk}-{mk}",
                        "Tip": "Finanțe",
                        "Venituri": income,
                        "Cheltuieli": expense,
                        "Profit": income - expense,
                        "Locatie": "",
                        "Comentariu": "",
                        "Detalii (Mat/Pers)": ""
                    })
                
                for e in entries:
                    row = {
                        "Data": f"{dk}-{mk}",
                        "Tip": e['Tip'],
                        "Venituri": "",
                        "Cheltuieli": "",
                        "Profit": "",
                        "Locatie": e.get('Locatie', ''),
                        "Comentariu": e.get('Comentariu', '')
                    }
                    row["Detalii (Mat/Pers)"] = self.format_materials(e).replace('\n', ' | ') if e['Tip'] == "Muncă" else e.get('Persoane', '')
                    rows.append(row)
        
        if rows:
            path = filedialog.asksaveasfilename(defaultextension=".xlsx", filetypes=[("Excel", "*.xlsx")])
            if path:
                pd.DataFrame(rows).to_excel(path, index=False)
                messagebox.showinfo("📈 Export", "Fișier Excel creat cu succes!")
        else:
            messagebox.showwarning("!", "Nu există date de exportat.")
